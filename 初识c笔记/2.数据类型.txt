打印每种类型的大小

#include <stdio.h>
int main()
{
	//printf("miku\n");
	//printf("初音未来\n");
	//printf("%d\n", 666 );//打印一个整数 - %d
	//sizeof - 关键字 - 操作符 - 计算类型或者所占空间的大小

	printf("%d\n", sizeof(char));
	printf("%d\n", sizeof(short));
	printf("%d\n", sizeof(int));
	printf("%d\n", sizeof(long));
	printf("%d\n", sizeof(long long));
	printf("%d\n", sizeof(float));
	printf("%d\n", sizeof(double));

	return 0;
}



这样也可
#include <stdio.h>
int main()
{
	printf("%zu\n", sizeof(char));               1个字节
	printf("%zu\n", sizeof(short));              2个字节
	printf("%zu\n", sizeof(int));                   4个字节
	printf("%zu\n", sizeof(long));                4个字节
	printf("%zu\n", sizeof(long long));       8个字节
	printf("%zu\n", sizeof(float));                4个字节
	printf("%zu\n", sizeof(double));            8个字节

	return 0;
}
